public class MainActivity extends AppCompatActivity {
    
    private Button button;
    private EditText editText;
    private CheckBox checkBox;
    private RadioGroup radioGroup;
    private Spinner spinner;
    
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        // Initialize views
        button = findViewById(R.id.buttonSubmit);
        editText = findViewById(R.id.editTextName);
        checkBox = findViewById(R.id.checkBoxAgree);
        radioGroup = findViewById(R.id.radioGroupGender);
        spinner = findViewById(R.id.spinnerCountry);
        
        // Button click listener
        button.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                handleButtonClick();
            }
        });
        
        // Lambda expression (modern approach)
        button.setOnClickListener(v -> handleButtonClick());
    }
    
    private void handleButtonClick() {
        String name = editText.getText().toString();
        boolean isAgreed = checkBox.isChecked();
        
        if (name.isEmpty()) {
            Toast.makeText(this, "Please enter your name", 
                         Toast.LENGTH_SHORT).show();
            return;
        }
        
        if (!isAgreed) {
            Toast.makeText(this, "Please agree to terms", 
                         Toast.LENGTH_SHORT).show();
            return;
        }
        
        Toast.makeText(this, "Registration successful!", 
                     Toast.LENGTH_LONG).show();
    }
}

        backButton.setOnClickListener(v -> {
            finish(); // Close current activity
        });
        
        // Result button click
        resultButton.setOnClickListener(v -> {
            Intent resultIntent = new Intent();
            resultIntent.putExtra("result_message", "Task completed successfully!");
            resultIntent.putExtra("completion_time", System.currentTimeMillis());
            setResult(RESULT_OK, resultIntent);
            finish();
        });
    }
    
    @Override
    public void onBackPressed() {
        // Handle back button press
        Intent resultIntent = new Intent();
        resultIntent.putExtra("result_message", "User pressed back button");
        setResult(RESULT_CANCELED, resultIntent);
        super.onBackPressed();
    }
}

private void sendDataToSecondActivity() {
    Intent intent = new Intent(MainActivity.this, SecondActivity.class);
    
    // Basic data types
    intent.putExtra("string_data", "Hello World");
    intent.putExtra("int_data", 42);
    intent.putExtra("boolean_data", true);
    intent.putExtra("double_data", 3.14159);
    intent.putExtra("long_data", System.currentTimeMillis());
    
    // Arrays
    String[] stringArray = {"Apple", "Banana", "Orange"};
    intent.putExtra("string_array", stringArray);
    
    int[] intArray = {1, 2, 3, 4, 5};
    intent.putExtra("int_array", intArray);
    
    // ArrayList
    ArrayList<String> stringList = new ArrayList<>();
    stringList.add("Item 1");
    stringList.add("Item 2");
    intent.putStringArrayListExtra("string_list", stringList);
    
    // Bundle for complex data
    Bundle bundle = new Bundle();
    bundle.putString("user_name", "John Doe");
    bundle.putInt("user_age", 25);
    bundle.putBoolean("is_verified", true);
    intent.putExtras(bundle);
    
    // Serializable object
    User user = new User("Jane", "jane@example.com", 30);
    intent.putExtra("user_object", user);
    
    startActivity(intent);
}

private void receiveDataFromIntent() {

        boolean booleanData = intent.getBooleanExtra("boolean_data", false);
        double doubleData = intent.getDoubleExtra("double_data", 0.0);
        long longData = intent.getLongExtra("long_data", 0L);
        
        // Arrays
        String[] stringArray = intent.getStringArrayExtra("string_array");
        int[] intArray = intent.getIntArrayExtra("int_array");
        
        // ArrayList
        ArrayList<String> stringList = intent.getStringArrayListExtra("string_list");

                       String name = editTextName.getText().toString().trim();
                       String email = editTextEmail.getText().toString().trim();
                       
                       if (validateInput(name, email)) {
                           listener.onDialogPositiveClick(name, email);
                       }
                   }
               })
               .setNegativeButton("Cancel", new DialogInterface.OnClickListener() {
                   @Override
                   public void onClick(DialogInterface dialog, int id) {
                       listener.onDialogNegativeClick();
                   }
               })
               .setNeutralButton("Clear", new DialogInterface.OnClickListener() {
                   @Override

                       clearFields();
                   }
               });
               
        return builder.create();
    }

    public void onStart() {

        // Customize dialog appearance
        Dialog dialog = getDialog();

                ViewGroup.LayoutParams.MATCH_PARENT,
                ViewGroup.LayoutParams.WRAP_CONTENT
            );
        }

            return false;
        }
        
        return true;
    }

        editTextEmail.setText("");
        checkBoxNewsletter.setChecked(false);

public class SpacingItemDecoration extends RecyclerView.ItemDecoration {

        this.spacing = spacing;
    }
    
    @Override
    public void getItemOffsets(@NonNull Rect outRect, @NonNull View view, 
                              @NonNull RecyclerView parent, @NonNull RecyclerView.State state) {
        outRect.left = spacing;
        outRect.right = spacing;
        outRect.bottom = spacing;
        
        // Add top margin only for the first item to avoid double spacing
        if (parent.getChildAdapterPosition(view) == 0) {
            outRect.top = spacing;
        }
    }

public class DatabaseHelper extends SQLiteOpenHelper {
    
    // Database Info
    private static final String DATABASE_NAME = "StudentDB";
    private static final int DATABASE_VERSION = 1;
    
    // Table Names
    private static final String TABLE_STUDENTS = "students";
    private static final String TABLE_COURSES = "courses";
    private static final String TABLE_ENROLLMENTS = "enrollments";
    
    // Students Table Columns
    private static final String KEY_STUDENT_ID = "student_id";
    private static final String KEY_STUDENT_NAME = "name";
    private static final String KEY_STUDENT_EMAIL = "email";
    private static final String KEY_STUDENT_PHONE = "phone";
    private static final String KEY_STUDENT_GRADE = "grade";
    private static final String KEY_STUDENT_CREATED = "created_at";
    
    // Courses Table Columns
    private static final String KEY_COURSE_ID = "course_id";
    private static final String KEY_COURSE_NAME = "course_name";
    private static final String KEY_COURSE_CODE = "course_code";
    private static final String KEY_COURSE_CREDITS = "credits";
    
    // Enrollments Table Columns
    private static final String KEY_ENROLLMENT_ID = "enrollment_id";
    private static final String KEY_ENROLLMENT_STUDENT_ID = "student_id";
    private static final String KEY_ENROLLMENT_COURSE_ID = "course_id";
    private static final String KEY_ENROLLMENT_DATE = "enrollment_date";
    
    public DatabaseHelper(Context context) {
        super(context, DATABASE_NAME, null, DATABASE_VERSION);
    }
    
    @Override
    public void onCreate(SQLiteDatabase db) {
        // Create Students table
        String CREATE_STUDENTS_TABLE = "CREATE TABLE " + TABLE_STUDENTS + "("
                + KEY_STUDENT_ID + " INTEGER PRIMARY KEY AUTOINCREMENT,"
                + KEY_STUDENT_NAME + " TEXT NOT NULL,"
                + KEY_STUDENT_EMAIL + " TEXT UNIQUE NOT NULL,"
                + KEY_STUDENT_PHONE + " TEXT,"
                + KEY_STUDENT_GRADE + " TEXT,"
                + KEY_STUDENT_CREATED + " DATETIME DEFAULT CURRENT_TIMESTAMP"
                + ")";
        
        // Create Courses table
        String CREATE_COURSES_TABLE = "CREATE TABLE " + TABLE_COURSES + "("
                + KEY_COURSE_ID + " INTEGER PRIMARY KEY AUTOINCREMENT,"
                + KEY_COURSE_NAME + " TEXT NOT NULL,"
                + KEY_COURSE_CODE + " TEXT UNIQUE NOT NULL,"
                + KEY_COURSE_CREDITS + " INTEGER DEFAULT 3"
                + ")";
        
        // Create Enrollments table
        String CREATE_ENROLLMENTS_TABLE = "CREATE TABLE " + TABLE_ENROLLMENTS + "("
                + KEY_ENROLLMENT_ID + " INTEGER PRIMARY KEY AUTOINCREMENT,"
                + KEY_ENROLLMENT_STUDENT_ID + " INTEGER,"
                + KEY_ENROLLMENT_COURSE_ID + " INTEGER,"
                + KEY_ENROLLMENT_DATE + " DATETIME DEFAULT CURRENT_TIMESTAMP,"
                + "FOREIGN KEY(" + KEY_ENROLLMENT_STUDENT_ID + ") REFERENCES " 
                + TABLE_STUDENTS + "(" + KEY_STUDENT_ID + "),"
                + "FOREIGN KEY(" + KEY_ENROLLMENT_COURSE_ID + ") REFERENCES " 
                + TABLE_COURSES + "(" + KEY_COURSE_ID + ")"
                + ")";
        
        db.execSQL(CREATE_STUDENTS_TABLE);
        db.execSQL(CREATE_COURSES_TABLE);
        db.execSQL(CREATE_ENROLLMENTS_TABLE);
        
        // Insert sample data
        insertSampleData(db);
    }
    
    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
        // Drop older tables if existed
        db.execSQL("DROP TABLE IF EXISTS " + TABLE_ENROLLMENTS);
        db.execSQL("DROP TABLE IF EXISTS " + TABLE_COURSES);
        db.execSQL("DROP TABLE IF EXISTS " + TABLE_STUDENTS);
        
        // Create tables again
        onCreate(db);
    }

        values.put("grade", student.getGrade());

    public List<Student> getAllStudents() {
        List<Student> students = new ArrayList<>();
        
        String selectQuery = "SELECT * FROM students ORDER BY name ASC";
        Cursor cursor = database.rawQuery(selectQuery, null);
        
        if (cursor.moveToFirst()) {
            do {
                Student student = new Student();
                student.setId(cursor.getString(cursor.getColumnIndex("student_id")));
                student.setName(cursor.getString(cursor.getColumnIndex("name")));
                student.setEmail(cursor.getString(cursor.getColumnIndex("email")));
                student.setPhone(cursor.getString(cursor.getColumnIndex("phone")));
                student.setGrade(cursor.getString(cursor.getColumnIndex("grade")));
                
                students.add(student);
            } while (cursor.moveToNext());
        }
        
        cursor.close();
        return students;
